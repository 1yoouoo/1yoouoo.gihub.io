N = int(input())
lists = []              # [5 2 3 4 1]
result = []             # [1 2 3 4 5]

for _ in range(N):
    A = int(input())
    lists.append(A)

def min_value(asd):       # 리스트 내 최솟값 구하기
    min_idx = 0
    M = len(lists)
    for i in range(M):
        if lists[i] < lists[min_idx]:
            min_idx = i

### 11650 좌표 정렬하기
```python
N = int(input())
lists = []
for i in range(N):
    [x, y] = map(int, input().split())
    lists.append([x, y])  

lists.sort()

for j in range(N):
    print(lists[j][0], lists[j][1])
```
> [x, y] = map(int, input().split()) 이 부분만 떠올리면 문제 없이 해결 할 수 있다.
